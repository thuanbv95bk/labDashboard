heightCard: number = 0;

  getHeightCard() {
    this.heightCard = this.getHeight('.chart-doughnut-container'); // Cập nhật giá trị
    console.log('height:', this.heightCard); // Giá trị cũ, chưa cập nhật

    return {
      height: `${this.heightCard}px`,
    };
  }

  getHeight(selectors: string): number {
    const elements = Array.from(
      document.querySelectorAll(selectors)
    ) as HTMLElement[];

    if (elements.length === 0) return 0; // Không có phần tử, trả về 0

    return Math.max(
      ...elements.map((el) => el.getBoundingClientRect().height || 0)
    );
  }
